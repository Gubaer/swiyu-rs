# Test commands for swiyu-issuer (HTTPie / `http` CLI).
#
# Prereqs:
#   docker compose up -d                  # Postgres on host port 5433
#   cargo run --bin swiyu-issuer-mgmtapi  # listens on 0.0.0.0:8080
#   cargo run --bin swiyu-issuer-oidcapi  # listens on 0.0.0.0:8081
#
# Tip: install HTTPie via your package manager (apt, brew, pacman, ...)
#      or `pip install httpie`. Pipe through `| jq .` for extra polish.
#
# Authentication: every /api/v1/... request needs an
# Authorization: Bearer tok_<...> header. The contributor mints their
# own dev token after bootstrapping their dev tenant:
#
#   swiyu-issuer-cli tenant bootstrap-dev-from-env   # creates tenant, prints id
#   swiyu-issuer-cli tenant api-token mint --tenant dev   # prints tok_... once
#
# Export the token + your issuer id for the snippets below:
#
#   TOK="tok_..."
#   ISSUER="<bare-issuer-id>"
#
# /healthz and /readyz are operational probes and are NOT
# authenticated.


# -----------------------------------------------------------------------------
# Sanity checks (unauthenticated)
# -----------------------------------------------------------------------------

http :8080/healthz
http :8080/readyz


# -----------------------------------------------------------------------------
# Happy path: create a credential offer
# -----------------------------------------------------------------------------
# Expected: 201 Created with { id, pre_auth_code, offer_deeplink, expires_at }.

http POST :8080/api/v1/issuers/$ISSUER/credential-offers \
  "Authorization:Bearer $TOK" \
  vct=urn:communal:local-residence-id \
  claims:='{
    "family_name": "Müller",
    "given_name": "Anna",
    "birth_date": "1990-04-12",
    "commune_bfs": 4003,
    "commune_name": "Buchs SG",
    "valid_until": "2026-12-31"
  }'


# -----------------------------------------------------------------------------
# Happy path: fetch the credential offer just created
# -----------------------------------------------------------------------------
# Replace <bare-offer-id> with the bare part of the `id` returned by
# the create above (the bit after `offer_`). Expected: 200 OK with
# the offer details. The `state` field is "pending" until the wallet
# picks it up, "expired" once `expires_at` has passed, "cancelled"
# after a cancel, "issued" after wallet redemption.

http :8080/api/v1/issuers/$ISSUER/credential-offers/<bare-offer-id> \
  "Authorization:Bearer $TOK"


# -----------------------------------------------------------------------------
# Cancel an offer
# -----------------------------------------------------------------------------
# Expected: 200 with the full offer body, state=cancelled.
# Idempotent: re-running on a cancelled offer returns the same body.
# A 409 is returned if the offer is already issued.

http POST :8080/api/v1/issuers/$ISSUER/credential-offers/<bare-offer-id>/cancel \
  "Authorization:Bearer $TOK"


# -----------------------------------------------------------------------------
# Lightweight status check
# -----------------------------------------------------------------------------
# Expected: 200 with { id, state, expires_at, issued_at, cancelled_at }.
# No claims, no PII; suitable for polling.

http :8080/api/v1/issuers/$ISSUER/credential-offers/<bare-offer-id>/status \
  "Authorization:Bearer $TOK"


# -----------------------------------------------------------------------------
# List offers (cursor-paginated, newest first)
# -----------------------------------------------------------------------------
# Expected: 200 with { items: [...], next_cursor: "..." | null }.

http :8080/api/v1/issuers/$ISSUER/credential-offers \
  "Authorization:Bearer $TOK" \
  limit==10

# Filter by observed state (pending | issued | cancelled | expired):
http :8080/api/v1/issuers/$ISSUER/credential-offers \
  "Authorization:Bearer $TOK" \
  state==pending


# -----------------------------------------------------------------------------
# Auth error paths
# -----------------------------------------------------------------------------

# Missing Authorization header -> 401 unauthorised
http :8080/api/v1/issuers/$ISSUER/credential-offers/zzzzzzzzzzzzzz

# Wrong scheme -> 401 unauthorised
http :8080/api/v1/issuers/$ISSUER/credential-offers/zzzzzzzzzzzzzz \
  "Authorization:Basic dXNlcjpwYXNz"

# Unknown token (right shape, wrong secret) -> 401 unauthorised
http :8080/api/v1/issuers/$ISSUER/credential-offers/zzzzzzzzzzzzzz \
  "Authorization:Bearer tok_NotARealTokenAtAll1NotARealTokenAtAll1"


# -----------------------------------------------------------------------------
# Application error paths (all require a valid Authorization header)
# -----------------------------------------------------------------------------

# Unknown vct -> 400 unknown_vct
http POST :8080/api/v1/issuers/$ISSUER/credential-offers \
  "Authorization:Bearer $TOK" \
  vct=urn:bogus \
  claims:='{}'

# Wrong issuer for this tenant -> 404 not_found
# (zzzzzzzzzzzzzz parses as a valid bare base58 id; persistence
# returns "not found" because no such issuer is registered for the
# seeded tenant.)
http POST :8080/api/v1/issuers/zzzzzzzzzzzzzz/credential-offers \
  "Authorization:Bearer $TOK" \
  vct=urn:communal:local-residence-id \
  claims:='{}'

# Malformed issuer id (non-base58 character) -> 400 invalid_input
http POST :8080/api/v1/issuers/notValid0/credential-offers \
  "Authorization:Bearer $TOK" \
  vct=urn:communal:local-residence-id \
  claims:='{}'

# Schema mismatch (missing required fields) -> 400 claims_validation_failed
http POST :8080/api/v1/issuers/$ISSUER/credential-offers \
  "Authorization:Bearer $TOK" \
  vct=urn:communal:local-residence-id \
  claims:='{"family_name": "Test"}'

# Get with unknown offer id (valid base58 but no such row) -> 404 not_found
http :8080/api/v1/issuers/$ISSUER/credential-offers/zzzzzzzzzzzzzz \
  "Authorization:Bearer $TOK"

# Get with malformed offer id (non-base58 character) -> 400 invalid_input
http :8080/api/v1/issuers/$ISSUER/credential-offers/notValid0 \
  "Authorization:Bearer $TOK"

# expires_in_seconds out of range -> 400 invalid_input
http POST :8080/api/v1/issuers/$ISSUER/credential-offers \
  "Authorization:Bearer $TOK" \
  vct=urn:communal:local-residence-id \
  expires_in_seconds:=10 \
  claims:='{
    "family_name": "Müller",
    "given_name": "Anna",
    "birth_date": "1990-04-12",
    "commune_bfs": 4003,
    "commune_name": "Buchs SG",
    "valid_until": "2026-12-31"
  }'


# -----------------------------------------------------------------------------
# Wallet flow (swiyu-issuer-oidcapi on :8081, no Authorization header)
# -----------------------------------------------------------------------------

# Issuer metadata
http :8081/i/$ISSUER/.well-known/openid-credential-issuer

# OAuth authorization-server metadata
http :8081/i/$ISSUER/.well-known/oauth-authorization-server

# Fetch the credential offer body (requires the offer to be pending and
# unexpired). Replace <bare-offer-id> with the bare part of the `id`
# returned by create above.
http :8081/i/$ISSUER/credential-offer/<bare-offer-id>

# Exchange a pre-authorised code for an access token + c_nonce. The
# bare pre-auth code is what the management API returned in the
# `pre_auth_code` field at offer creation. Form-encoded body; note the
# hyphen in `pre-authorized_code`.
http --form POST :8081/i/$ISSUER/token \
  grant_type=urn:ietf:params:oauth:grant-type:pre-authorized_code \
  pre-authorized_code=<bare-pre-auth-code>

# Replaying the same pre-auth code -> 400 invalid_grant (the unique
# constraint on oidc_access_tokens.offer_id rejects the second insert).
http --form POST :8081/i/$ISSUER/token \
  grant_type=urn:ietf:params:oauth:grant-type:pre-authorized_code \
  pre-authorized_code=<bare-pre-auth-code>

# Wrong grant type -> 400 unsupported_grant_type
http --form POST :8081/i/$ISSUER/token \
  grant_type=client_credentials \
  pre-authorized_code=anything

# Issue the credential. Requires:
#   <access-token>: from /token's response
#   <wallet-proof-jwt>: a wallet-signed JWT with header.jwk = wallet
#                       public key, payload.aud = ISSUER_BASE_URL/i/<issuer-id>,
#                       payload.iat within +/- 5 minutes, payload.nonce =
#                       the c_nonce returned by /token.
# The proof JWT signature is verified against the embedded `jwk` (ES256
# or EdDSA). The issued credential is signed with the issuer's assertion
# key via the configured SIGNING_ENGINE (dev: DB-backed; vault: HashiCorp).
http POST :8081/i/$ISSUER/credential \
  "Authorization:Bearer <access-token>" \
  format=vc+sd-jwt \
  vct=urn:communal:local-residence-id \
  proof:='{"proof_type": "jwt", "jwt": "<wallet-proof-jwt>"}'

# Mint a fresh token (CLI; no HTTP)
# -----------------------------------------------------------------------------
# Prints id/name/secret once; only the hash is persisted.
#   cargo run --bin swiyu-issuer-cli -- tenant api-token mint --tenant dev
#   cargo run --bin swiyu-issuer-cli -- tenant api-token mint \
#     --tenant dev --name short-lived --expires-in 1h
#
# With --token-only it prints just the bare tok_... for scripting:
#   TOK=$(cargo run -q --bin swiyu-issuer-cli -- \
#     tenant api-token mint --tenant dev --token-only)
#
# --tenant also accepts a bare tenant id or a business partner UUID;
# --name is optional and defaults to a generated label.
